CALENDARIO = 
      VAR DIA_INI = "01" 
      VAR DIA_FIM = "31" 
      VAR MES_INI = MONTH(FIRSTDATE(FATO_VENDA[DATA_EMISSAO]))
      VAR ANO_INI = YEAR(FIRSTDATE(FATO_VENDA[DATA_EMISSAO]))
      VAR MES_FIM = MONTH(LASTDATE(FATO_META[DATA_REF]))
      VAR ANO_FIM = YEAR (LASTDATE(FATO_META[DATA_REF]))
      RETURN 
      CALENDAR(DATE(ANO_INI;MES_INI;DIA_INI);DATE(ANO_FIM;MES_FIM;DIA_FIM))
